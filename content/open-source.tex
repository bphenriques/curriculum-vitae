    {\href{https://github.com/bphenriques/selfhost-nix}{bphenriques/selfhost-nix}}
    {January 2025 - Present}
    {
    \begin{openSourceItems}
        \item \textbf{Reusable NixOS framework} for self-hosting on an interface/implementation (provider-swappable) architecture --- one service declaration wires ingress + TLS, SSO/OIDC, metrics, backups, and notifications.
        \item \textbf{Security-first:} secrets generated outside the Nix store with rotation, auto-provisioned per-service OIDC clients, and hardened systemd units.
        \item \textbf{Shipped as a real project:} NixOS VM integration tests in CI, an mdBook docs site, and a \texttt{nix flake init} consumer template.
    \end{openSourceItems}
    \vspace{4mm}
    \skilltaglist{\textbf{Nix},\textbf{NixOS},\textbf{OIDC/SSO},Traefik,Prometheus,systemd,WireGuard,sops-nix,Shell Scripting}
